const express = require('express');
const app = express();
const PORT = 8080;

app.listen(PORT , ()=>{
	console.log("Express is running");
});

app.get('/' , (req,res)=>{
	res.send('Home')
});

app.get('/Homepage' , (req,res)=>{
	res.send('Homepage');
});

app.get('/Contactus' , (req,res)=>{
	res.send('Contact us');
});

app.get('/Aboutus' , (req,res)=>{
	res.send('About us');
});
